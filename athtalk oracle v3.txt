            mathTerms: ['inverse', 'coefficient', 'divisor', 'isolate', 'factor', 'product', 'quotient', 'sum', 'difference', 'ratio', 'proportion'],
            academicVerbs: ['explain', 'justify', 'compare', 'analyze', 'evaluate', 'describe', 'determine']

        // Load data from localStorage
        function loadData() {
            try {
                const stored = localStorage.getItem('mathtalk_v3_students');
                if (stored) state.students = JSON.parse(stored);

                const dataMap = localStorage.getItem('mathtalk_v3_data_map');
                if (dataMap) {
                    state.studentDataMap = JSON.parse(dataMap);
                } else {
                    state.studentDataMap = sampleData;
                    saveData();
                }

                const lesson = localStorage.getItem('mathtalk_v3_lesson');
                if (lesson) state.currentLesson = JSON.parse(lesson);

                const writing = localStorage.getItem('mathtalk_v3_writing');
                if (writing) state.writingData = JSON.parse(writing);
            } catch (e) {

            }
        }

        // Save data to localStorage
        function saveData() {
            localStorage.setItem('mathtalk_v3_students', JSON.stringify(state.students));
            localStorage.setItem('mathtalk_v3_data_map', JSON.stringify(state.studentDataMap));
            if (state.currentLesson) localStorage.setItem('mathtalk_v3_lesson', JSON.stringify(state.currentLesson));
            localStorage.setItem('mathtalk_v3_writing', JSON.stringify(state.writingData));
        }

        // Analyze written response with TWR awareness
        function analyzeWriting(text, studentData) {
            const analysis = {
                sentenceCount: 0,
                wordCount: 0,
                twrConnectors: [],
                mathVocabUsed: [],
                academicVerbsUsed: [],
                sentenceComplexity: 0,
                vocabularyScore: 0,
                grammarScore: 0,
                twrScore: 0,
                overallWritingScore: 0,
                feedback: [],
                englishTips: []
            };

            const lowerText = text.toLowerCase();
            const sentences = text.split(/[.!?]+/).filter(s => s.trim().length > 5);
            analysis.sentenceCount = sentences.length;
            analysis.wordCount = text.split(/\s+/).length;

            // Check for TWR connectors
            twrVocab.connectors.forEach(word => {
                if (lowerText.includes(word)) {
                    analysis.twrConnectors.push(word);
                }
            });

            // Check for math vocabulary
            twrVocab.mathTerms.forEach(term => {
                if (lowerText.includes(term)) {
                    analysis.mathVocabUsed.push(term);
                }
            });

            // Check for academic verbs
            twrVocab.academicVerbs.forEach(verb => {
                if (lowerText.includes(verb)) {
                    analysis.academicVerbsUsed.push(verb);
                }
            });

            // Sentence complexity (average words per sentence)
            if (analysis.sentenceCount > 0) {
                analysis.sentenceComplexity = Math.round(analysis.wordCount / analysis.sentenceCount);
            }

            // Vocabulary score (0-100)
            const vocabUsage = analysis.mathVocabUsed.length + analysis.academicVerbsUsed.length;
            analysis.vocabularyScore = Math.min(100, vocabUsage * 25);

            // TWR score (0-100)
            analysis.twrScore = Math.min(100, analysis.twrConnectors.length * 33);

            // Grammar score (simplified - based on sentence structure)
            const hasCapitalization = /[A-Z]/.test(text);
            const hasPunctuation = /[.!?]/.test(text);
            const hasCompleteThought = analysis.wordCount > 8;
            analysis.grammarScore = (hasCapitalization ? 30 : 0) + (hasPunctuation ? 30 : 0) + (hasCompleteThought ? 40 : 0);

            // Overall writing score
            analysis.overallWritingScore = Math.round(
                (analysis.vocabularyScore * 0.35) + 
                (analysis.twrScore * 0.35) + 
                (analysis.grammarScore * 0.30)
            );

            // Generate feedback
            if (analysis.twrConnectors.length === 0) {
                analysis.feedback.push("Try using connecting words like 'because' or 'so' to explain your thinking.");
                analysis.englishTips.push("'Because' helps you give reasons: 'I chose this answer because...'");
            } else if (analysis.twrConnectors.length === 1) {
                analysis.feedback.push(`Great use of "${analysis.twrConnectors[0]}"! Can you also use "therefore" or "however"?`);
            } else {
                analysis.feedback.push(`Excellent! You used ${analysis.twrConnectors.length} connecting words.`);
            }

            if (analysis.mathVocabUsed.length === 0) {
                analysis.feedback.push("Try using math vocabulary like 'factor', 'quotient', or 'coefficient'.");
            } else {
                analysis.feedback.push(`Strong math vocabulary! You used: ${analysis.mathVocabUsed.join(', ')}`);
            }

            // English learning tips based on WIDA level
            if (studentData && studentData.wida < 3.0) {
                if (analysis.sentenceComplexity < 8) {
                    analysis.englishTips.push("Tip: Try adding more details. Instead of 'I solved it', write 'I solved it by multiplying both sides'");
                }
            }

            return analysis;
        }

        // Get language complexity level
        function getLanguageComplexity(widaLevel) {
            if (widaLevel <= 2.0) return 'entering';
            if (widaLevel <= 3.0) return 'emerging';
            if (widaLevel <= 4.0) return 'developing';
            if (widaLevel <= 5.0) return 'expanding';
            return 'bridging';
        }

        // Generate AI response with Claude API
        async function generateAIResponse(userMessage, conversationHistory, studentData) {
            state.isLoading = true;
            render();

            try {
                const widaLevel = studentData?.wida || 3.0;
                const complexity = getLanguageComplexity(widaLevel);

                let languageGuidance = '';
                if (complexity === 'entering') {
                    languageGuidance = 'Use VERY simple 5-8 word sentences. Ask yes/no questions. Teach key English words.';
                } else if (complexity === 'emerging') {
                    languageGuidance = 'Use simple 8-12 word sentences. Provide sentence frames like "I think ___ because ___". Teach vocabulary.';
                } else if (complexity === 'developing') {
                    languageGuidance = 'Use clear academic language. Encourage use of "because/so/but". Correct grammar gently.';
                } else {
                    languageGuidance = 'Use grade-level academic language. Push for precise explanations with TWR structures.';
                }

                const systemPrompt = `You are MathTalk Oracle, a 6th-grade math tutor focused on BOTH math understanding AND English language development.

STUDENT PROFILE:
- WIDA Level: ${widaLevel} (${complexity})
- Language Guidance: ${languageGuidance}

YOUR DUAL ROLE:
1. Help student explain math thinking
2. Teach English vocabulary and sentence structures

WRITING REVOLUTION (TWR) FOCUS:
- Encourage students to use: because, so, but, therefore, however
- Model academic sentence structures
- Praise use of math vocabulary: coefficient, inverse, quotient, factor, etc.

TEACH ENGLISH NATURALLY:
- If student makes grammar error, model correct form in your response
- Introduce 1-2 new vocabulary words per conversation
- Use sentence frames for emerging students

Ask ONE question (10-15 words max) that helps student explain AND practice English.

Respond with ONLY your question, nothing else.`;

                const response = await fetch('https://api.anthropic.com/v1/messages', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        model: 'claude-sonnet-4-20250514',
                        max_tokens: 1000,
                        system: systemPrompt,
                        messages: [
                            { role: 'user', content: `Student said: "${userMessage}"\n\nAsk your next question (include gentle English teaching if needed):` }
                        ]
                    })
                });

                const data = await response.json();
                return data.content.find(item => item.type === 'text')?.text || "Can you explain more?";
            } catch (error) {
                console.error('API Error:', error);
                return "Tell me more about your thinking.";
            } finally {
                state.isLoading = false;
            }
        }

        // Handle student message
        async function handleStudentMessage() {
            if (!state.currentInput.trim()) return;

            const message = state.currentInput.trim();
            const studentData = state.studentDataMap[state.currentStudent.username];
            
            // Analyze writing
            const writingAnalysis = analyzeWriting(message, studentData);

            // Add to conversation
            state.conversationHistory.push({
                role: 'student',
                content: message,
                writingAnalysis,
                timestamp: new Date().toISOString()
            });

            // Track writing progress
            if (!state.writingData[state.currentStudent.username]) {
                state.writingData[state.currentStudent.username] = {
                    analyses: [],
                    averageScores: { vocabulary: 0, twr: 0, grammar: 0, overall: 0 }
                };
            }
            state.writingData[state.currentStudent.username].analyses.push(writingAnalysis);

            // Calculate averages
            const analyses = state.writingData[state.currentStudent.username].analyses;
            const avgVocab = analyses.reduce((sum, a) => sum + a.vocabularyScore, 0) / analyses.length;
            const avgTWR = analyses.reduce((sum, a) => sum + a.twrScore, 0) / analyses.length;
            const avgGrammar = analyses.reduce((sum, a) => sum + a.grammarScore, 0) / analyses.length;
            const avgOverall = analyses.reduce((sum, a) => sum + a.overallWritingScore, 0) / analyses.length;

            state.writingData[state.currentStudent.username].averageScores = {
                vocabulary: Math.round(avgVocab),
                twr: Math.round(avgTWR),
                grammar: Math.round(avgGrammar),
                overall: Math.round(avgOverall)
            };

            state.currentInput = '';
            render();

            // Get AI response
            const aiResponse = await generateAIResponse(message, state.conversationHistory, studentData);
            
            state.conversationHistory.push({
                role: 'ai',
                content: aiResponse,
                timestamp: new Date().toISOString()
            });

            // Speak AI response if voice is enabled
            if (voiceEnabled) {
                speakText(aiResponse);
            }

            // Update student record
            const updatedStudents = state.students.map(s => {
                if (s.username === state.currentStudent.username) {
                    return {
                        ...s,
                        conversations: [...(s.conversations || []), {
                            timestamp: new Date().toISOString(),
                            messages: state.conversationHistory
                        }]
                    };
                }
                return s;
            });

            state.students = updatedStudents;
            state.currentStudent = updatedStudents.find(s => s.username === state.currentStudent.username);
            saveData();
            render();
        }

        // Voice synthesis
        let voiceEnabled = false;
        let recognition = null;
        let synthesis = window.speechSynthesis;

        function toggleVoice() {
            voiceEnabled = !voiceEnabled;
            render();
        }

        function startVoiceInput() {
            if ('webkitSpeechRecognition' in window || 'SpeechRecognition' in window) {
                const SpeechRecognition = window.webkitSpeechRecognition || window.SpeechRecognition;
                recognition = new SpeechRecognition();
                recognition.continuous = false;
                recognition.interimResults = false;
                recognition.lang = 'en-US';

                recognition.onresult = (event) => {
                    const transcript = event.results[0][0].transcript;
                    state.currentInput = transcript;
                    render();
                    document.getElementById('messageInput').value = transcript;
                };

                recognition.onerror = (event) => {
                    console.error('Speech recognition error:', event.error);
                };

                recognition.start();
            } else {
                alert('Voice input not supported in this browser. Try Chrome or Edge.');
            }
        }

        function speakText(text) {
            if (voiceEnabled && synthesis) {
                synthesis.cancel(); // Cancel any ongoing speech
                const utterance = new SpeechSynthesisUtterance(text);
                utterance.rate = 0.9;
                utterance.pitch = 1.0;
                utterance.volume = 1.0;
                synthesis.speak(utterance);
            }
        }

        // Export conversation transcript
        function exportConversation(studentName) {
            const student = state.students.find(s => s.username === studentName);
            if (!student || !student.conversations || student.conversations.length === 0) {
                alert('No conversations to export');
                return;
            }

            let text = `MathTalk Oracle - Conversation Transcript\n`;
            text += `Student: ${studentName}\n`;
            text += `Date: ${new Date().toLocaleDateString()}\n`;
            text += `\n${'='.repeat(60)}\n\n`;

            student.conversations.forEach((conv, i) => {
                text += `CONVERSATION ${i + 1} - ${new Date(conv.timestamp).toLocaleString()}\n`;
                text += `${'-'.repeat(60)}\n\n`;

                conv.messages.forEach(msg => {
                    if (msg.role === 'student') {
                        text += `STUDENT:\n${msg.content}\n`;
                        if (msg.writingAnalysis) {
                            text += `[Writing Score: ${msg.writingAnalysis.overallWritingScore}% | `;
                            text += `TWR: ${msg.writingAnalysis.twrConnectors.join(', ') || 'none'}]\n`;
                        }
                        text += `\n`;
                    } else {
                        text += `AI TUTOR:\n${msg.content}\n\n`;
                    }
                });

                text += `\n`;
            });

            const blob = new Blob([text], { type: 'text/plain' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `${studentName.replace(/\s+/g, '_')}_transcript.txt`;
            a.click();
        }

        function render() {
            const app = document.getElementById('app');
            let html = '';

            if (state.view === 'login') html = renderLogin();
            else if (state.view === 'student') html = renderStudent();
            else if (state.view === 'teacher') html = renderTeacher();

            app.innerHTML = html;

            // Update input value if exists
            if (state.view === 'student' && state.conversationHistory.length > 0) {
                const input = document.getElementById('messageInput');
                if (input) {
                    input.value = state.currentInput;
                    input.addEventListener('input', (e) => { state.currentInput = e.target.value; });
                }
            }

            if (state.view === 'login') {
                const input = document.getElementById('username');
                if (input) {
                    input.addEventListener('input', (e) => { state.username = e.target.value; });
                }
            }
        }

        // Event handlers
        function handleLogin(type) {
            state.username = document.getElementById('username')?.value.trim() || state.username;
            if (!state.username) return;

            state.userType = type;

            if (type === 'student') {
                let student = state.students.find(s => s.username === state.username);
                if (!student) {
                    student = {
                        username: state.username,
                        conversations: [],
                        createdAt: new Date().toISOString()
                    };
                    state.students.push(student);
                    saveData();
                }
                state.currentStudent = student;
                state.view = 'student';
                state.conversationHistory = [];
            } else {
                state.view = 'teacher';
            }

            render();
        }

        function handleLogout() {
            state.view = 'login';
            state.username = '';
            state.currentStudent = null;
            state.conversationHistory = [];
            state.currentInput = '';
            render();
        }

        async function startConversation() {
            const studentData = state.studentDataMap[state.currentStudent.username];
            const aiMessage = await generateAIResponse("Starting conversation", [], studentData);
            
            state.conversationHistory.push({
                role: 'ai',
                content: aiMessage,
                timestamp: new Date().toISOString()
            });
            
            render();
        }

        function showImportModal() {
            document.getElementById('importModal').classList.add('active');
        }

        function closeImportModal() {
            document.getElementById('importModal').classList.remove('active');
        }

        function handleImport() {
            const data = document.getElementById('importData').value;
            const lines = data.trim().split('\n');
            
            lines.forEach(line => {
                const parts = line.split('\t');
                if (parts.length >= 6) {
                    const [name, wida, iReady, unit1, unit2, unit3, ...notesParts] = parts;
                    state.studentDataMap[name.trim()] = {
                        wida: parseFloat(wida) || 3.0,
                        iReady: parseInt(iReady) || 460,
                        unit1: parseInt(unit1) || 0,
                        unit2: parseInt(unit2) || 0,
                        unit3: parseInt(unit3) || 0,
                        notes: notesParts.join(' ').trim()
                    };
                }
            });

            saveData();
            closeImportModal();
            render();
        }
